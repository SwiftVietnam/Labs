struct ContentView: View {
    var body: some View {
        List {
            NewsItemView(title: "Bản tin Swift #5")
            NewsItemView(title: "Bản tin Swift #4")
            NewsItemView(title: "Bản tin Swift #3")
            NewsItemView(title: "Bản tin Swift #2")

        }

struct NewsItemView: View {
    var title: String

    var body: some View {
        Text(title)
            .font(.headline)
    }
}


struct ContentView: View {

        "Bản tin Swift #4",
        "Bản tin Swift #3",
        "Bản tin Swift #2",
        "Bản tin Swift #1"
    ]
    var body: some View {
        List {
